```cpp
class Solution {
public:
    void sortColors(vector<int>& nums) {
        //  sort(nums.begin(), nums.end());
        int g = 0, b = nums.size()-1;
        for(int r = 0; r<=b; r++){
            if(nums[r]==0){
                swap(nums[r], nums[g++]);
            }
            else if(nums[r]==2){
                swap(nums[r], nums[b--]);
                r--;
            }
        }
    }
};
```
